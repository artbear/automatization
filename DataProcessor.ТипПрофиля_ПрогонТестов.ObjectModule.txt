Процедура ЗапуститьБазуВРежимеПредприятия(ПутьК1С, КаталогИБ, ПользовательИБ, ПарольИБ, ПараметрыЗапуска="")
	
	Команда = "%ПутьК1С%   ENTERPRISE /DisableStartupMessages  %КаталогИБ% /N %ПользовательИБ% /P %ПарольИБ% "
		+" %ПараметрыЗапуска% ";
		
	Команда = СтрЗаменить(Команда, "%ПутьК1С%", ПутьК1С);
	
	Команда = СтрЗаменить(Команда, "%КаталогИБ%", КаталогИБ);
	Команда = СтрЗаменить(Команда, "%ПользовательИБ%", ПользовательИБ);
	Команда = СтрЗаменить(Команда, "%ПарольИБ%", ПарольИБ);
	
	Команда = СтрЗаменить(Команда, "%ПараметрыЗапуска%", ПараметрыЗапуска);
	
	ПакетныйРежим.ВыполнитьИзКоманднойСтроки(Команда);
